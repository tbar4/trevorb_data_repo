31/// Specify when an item is available.
32pub enum ProtectionMode {
33    /// The data in the keychain can only be accessed when the device is
34    /// unlocked. Only available if a passcode is set on the device.
35    AccessibleWhenPasscodeSetThisDeviceOnly,
36    ///The data in the keychain item can be accessed only while the device is
37    /// unlocked by the user.
38    AccessibleWhenUnlockedThisDeviceOnly,
39    /// The data in the keychain item can be accessed only while the device is
40    /// unlocked by the user.
41    AccessibleWhenUnlocked,
42    /// The data in the keychain item cannot be accessed after a restart until
43    /// the device has been unlocked once by the user.
44    AccessibleAfterFirstUnlockThisDeviceOnly,
45    /// The data in the keychain item cannot be accessed after a restart until
46    /// the device has been unlocked once by the user.
47    AccessibleAfterFirstUnlock,
48}
49
50impl SecAccessControl {
51    /// Create `AccessControl` object from flags
52    pub fn create_with_flags(flags: CFOptionFlags) -> Result<Self> {
53        Self::create_with_protection(None, flags)
54    }
55
56    /// Create `AccessControl` object from a protection value and flags.
57    pub fn create_with_protection(protection: Option<ProtectionMode>, flags: CFOptionFlags) -> Result<Self> {
58        let protection_val = protection.map(|v| {
59            match v {
60                ProtectionMode::AccessibleWhenPasscodeSetThisDeviceOnly => unsafe { CFString::wrap_under_get_rule(kSecAttrAccessibleWhenPasscodeSetThisDeviceOnly) },
61                ProtectionMode::AccessibleWhenUnlockedThisDeviceOnly => unsafe { CFString::wrap_under_get_rule(kSecAttrAccessibleWhenUnlockedThisDeviceOnly) },
62                ProtectionMode::AccessibleWhenUnlocked => unsafe { CFString::wrap_under_get_rule(kSecAttrAccessibleWhenUnlocked) },
63                ProtectionMode::AccessibleAfterFirstUnlockThisDeviceOnly => unsafe { CFString::wrap_under_get_rule(kSecAttrAccessibleAfterFirstUnlockThisDeviceOnly) },
64                ProtectionMode::AccessibleAfterFirstUnlock => unsafe { CFString::wrap_under_get_rule(kSecAttrAccessibleAfterFirstUnlock) },
65            }
66        });
67        unsafe {
68            let access_control = SecAccessControlCreateWithFlags(
69                kCFAllocatorDefault,
70                protection_val.map(|v| v.as_CFTypeRef()).unwrap_or(null()),
71                flags,
72                ptr::null_mut(),
73            );
74            if access_control.is_null() {
75                Err(Error::from_code(errSecParam))
76            } else {
77                Ok(Self::wrap_under_create_rule(access_control))
78            }
79        }
80    }
81}
